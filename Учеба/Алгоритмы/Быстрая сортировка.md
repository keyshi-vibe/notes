#include <iostream>
#include <vector>
#include <algorithm> // для std::swap

int partition(std::vector<int>& arr, int low, int high) {
    int pivot = arr[high]; // Выбираем последний элемент как опорный
    int i = low - 1;       // Индекс для элементов меньше pivot

    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            std::swap(arr[i], arr[j]);
        }
    }
    // Ставим pivot на его законное место
    std::swap(arr[i + 1], arr[high]);
    return i + 1; // Возвращаем индекс pivot
}

void quickSort(std::vector<int>& arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high); // Индекс разделения

        quickSort(arr, low, pi - 1);  // Сортируем элементы до pivot
        quickSort(arr, pi + 1, high); // Сортируем элементы после pivot
    }
}
